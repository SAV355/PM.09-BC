const mongoose = require('mongoose');
const Calculation = require('../models/Calculation');
const fs = require('fs');
const path = require('path');
const { Transform } = require('stream');
const json2csv = require('json2csv').parse;
require('dotenv').config();

const connectDB = async () => {
    try {
        await mongoose.connect(process.env.MONGODB_URI || 'mongodb://localhost:27017/bank-calculator', {
            useNewUrlParser: true,
            useUnifiedTopology: true,
        });
        console.log('✅ MongoDB connected');
    } catch (error) {
        console.error('❌ MongoDB connection error:', error);
        process.exit(1);
    }
};

// Экспортирует в JSON Lines (формат, для Compass)
const exportToJSONLines = async (calculations, exportDir, timestamp) => {
    const filepath = path.join(exportDir, `calculations_export_${timestamp}.json`);
    const writeStream = fs.createWriteStream(filepath);

    for (const calc of calculations) {
// Преобразует ObjectId и даты в строки
        const doc = {
            ...calc,
            _id: calc._id.toString(),
            createdAt: calc.createdAt.toISOString(),
            updatedAt: calc.updatedAt.toISOString(),
        };
// Убирает поле __v, если оно есть
        delete doc.__v;
        writeStream.write(JSON.stringify(doc) + '\n');
    }

    writeStream.end();
    return new Promise((resolve, reject) => {
        writeStream.on('finish', () => {
            console.log(`✅ JSON Lines export saved to: ${filepath}`);
            resolve(filepath);
        });
        writeStream.on('error', reject);
    });
};

// Экспорт в CSV (альтернативный вариант)
const exportToCSV = async (calculations, exportDir, timestamp) => {
    const fields = [
        '_id', 'type', 'loanAmount', 'monthlyPayment', 'totalPayment',
        'overpayment', 'annualRate', 'termYears', 'userEmail', 'createdAt',
        'propertyType', 'carType', 'loanPurpose', 'currentAge', 'retirementAge'
    ];
    const opts = { fields };

    try {
        const csv = json2csv(calculations.map(calc => ({
            ...calc,
            _id: calc._id.toString(),
            createdAt: calc.createdAt.toISOString(),
        })), opts);

        const filepath = path.join(exportDir, `calculations_export_${timestamp}.csv`);
        fs.writeFileSync(filepath, csv);
        console.log(`✅ CSV export saved to: ${filepath}`);
        return filepath;
    } catch (err) {
        console.error('CSV conversion error:', err);
    }
};

const exportToJSON = async () => {
    try {
        const calculations = await Calculation.find({}).lean();

        const exportDir = path.join(__dirname, 'exports');
        if (!fs.existsSync(exportDir)) {
            fs.mkdirSync(exportDir, { recursive: true });
        }

        const timestamp = new Date().toISOString().replace(/[:.]/g, '-');

// Экспорт в JSON Lines (для Compass)
        await exportToJSONLines(calculations, exportDir, timestamp);

// Экспорт в CSV (опционально)
        await exportToCSV(calculations, exportDir, timestamp);

        console.log(`📊 Total records: ${calculations.length}`);
        process.exit(0);
    } catch (error) {
        console.error('❌ Error exporting data:', error);
        process.exit(1);
    }
};

const run = async () => {
    await connectDB();
    await exportToJSON();
};

run();
